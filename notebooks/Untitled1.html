input_dim = 2
output_dim = 2

# Initialization /placeholder values
X = T.dtensor3('X')

# generate data
input_var = theano.shared(np.asarray(train_input).astype(np.float64))
target_var = theano.shared(np.asarray(train_output).astype(np.float64))

# Reference
# From paper :IMPROVING PERFORMANCE OF RECURRENT NEURAL NETWORK WITH RELU NONLINEARITY
def norm_positive_definite(r):
    A = np.dot(r, r.transpose())/hidden_dim
    values, vectors = np.linalg.eig(A)
    e = np.amax(values)
    Whh = A/e
    return Whh

def norm_weight(n_in, n_out):
    low = 4 *-np.sqrt(6.0/(n_in + n_out))
    high = 4 *np.sqrt(6.0/(n_in + n_out))
    W = np.random.uniform(low=low, high=high, size=(n_in,n_out))
    return W
# weights of layer
# initialize weights with random values
init_1 = norm_weight(input_dim, hidden_dim)
init_2 = norm_weight(hidden_dim, output_dim)
init_h = norm_positive_definite(np.random.normal
    (0, 1, (hidden_dim, hidden_dim)))


W1 = theano.shared(init_1.astype(theano.config.floatX))
W2 = theano.shared(init_2.astype(theano.config.floatX))
Wh = theano.shared(init_h.astype(theano.config.floatX))

# layer 1 which creates recurrence
L1 = theano.shared(np.zeros((8, hidden_dim)).astype(theano.config.floatX))

# define step
def step(v, l_tm1, W1, W2, Wh):
    layer1 = T.nnet.relu(T.dot(v, W1) + T.dot(l_tm1, Wh))
    output_layer = T.nnet.sigmoid(T.dot(layer1, W2))
    return [layer1, output_layer]


# Computational graph
[layer1s, predictions], updates = theano.scan(fn=step,
                                        sequences=dict(input=X,taps=[0]),
                                        outputs_info=[L1, None],  # same format as step function
                                        non_sequences=[W1, W2, Wh])

def build_rnn():
    print("Building network ...")

    W1=pm.Normal('w_in_1', 0, sd=1,
                             shape=(input_dim, hidden_dim),
                             testval=init_1.astype(np.float64))
    # Weights from 1st to 2nd layer
    W2=pm.Normal('w_2_out', 0, sd=1,
                            shape=(hidden_dim, output_dim),
                            testval=init_2.astype(np.float64))

    # Weights from hidden layer to output
    Wh=pm.Normal('w_1_2', 0, sd=1,
                              shape=(hidden_dim,hidden_dim),
                              testval=init_h.astype(np.float64))

    prediction = theano.function(inputs = [X], outputs = predictions)


    return prediction

with pm.Model() as rnn:
    prediction = build_rnn()
    pres = prediction(input_var.get_value())
    out = pm.Bernoulli('out',pres,observed=target_var)
